	void trans(unsigned char *s1,char *s2){
		int i, n = 0, temp;
		for(i = 0; i < 74; i++){
			
			if(i==20 || i==22 || i==24 || i==26 || i==47 || i==49 || i==51 || i==53){
				s2[i*2] = '\0';
				
			}else{
				temp = n;
				if(temp >= 23 && temp <= 42){
					
					temp = 23 + (42 - temp);
					
				}
				switch(s1[temp]){
					
					case 0:
						/* 普通：○ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xf0;
						break;
					case 1:
						/* 炸弹：▲ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xf8;
						break;
					case 2:
						/* 钻石：◆ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xf4;
						break;
					case 3:
						/* 魔法：★ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xef;
						break;
					case 4:
						/* A：♀ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xe2;
						break;
					case 5:
						/* B：♂ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xe1;
						break;
					case 6:
						/* AB: ∏ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xc7;
						break;
					default:
						/* 终点：◎ */
						s2[i*2] = 0xa1;
						s2[i*2+1] = 0xf2;
						break;
				}
				
				n++;
			}
			
		}
		
		
		return;
	}
